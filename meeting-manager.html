📄 Supabase SQL セットアップ（初回のみ実行）

Supabase の SQL Editor で以下を実行してください：

-- ミーティングテーブル CREATE TABLE meetings ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, office text NOT NULL CHECK (office IN ('akita', 'yokote')), meeting_date date NOT NULL, note text, created_at timestamptz DEFAULT now() ); -- メンバー作業記録テーブル CREATE TABLE meeting_members ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, meeting_id uuid REFERENCES meetings(id) ON DELETE CASCADE, member_name text NOT NULL, current_work text, progress int CHECK (progress BETWEEN 0 AND 100), next_plan text ); -- 全員が読み書き可能（RLS無効化 or 適宜設定） ALTER TABLE meetings ENABLE ROW LEVEL SECURITY; ALTER TABLE meeting_members ENABLE ROW LEVEL SECURITY; CREATE POLICY "allow_all_meetings" ON meetings FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all_members" ON meeting_members FOR ALL USING (true) WITH CHECK (true);
