Supabase の SQL Editor で以下を実行してください：

-- ミーティングテーブル CREATE TABLE meetings ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, office text NOT NULL CHECK (office IN ('akita', 'yokote')), meeting_date date NOT NULL, note text, created_at timestamptz DEFAULT now() ); CREATE TABLE meeting_members ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, meeting_id uuid REFERENCES meetings(id) ON DELETE CASCADE, member_name text NOT NULL, current_work text, progress int CHECK (progress BETWEEN 0 AND 100), next_plan text ); CREATE TABLE notices ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, content text NOT NULL, created_at timestamptz DEFAULT now() ); ALTER TABLE meetings ENABLE ROW LEVEL SECURITY; ALTER TABLE meeting_members ENABLE ROW LEVEL SECURITY; ALTER TABLE notices ENABLE ROW LEVEL SECURITY; CREATE POLICY "allow_all_meetings" ON meetings FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all_members" ON meeting_members FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "allow_all_notices" ON notices FOR ALL USING (true) WITH CHECK (true);
